// 获取IP
async function getIp() {
  try {
    let res = await fetch('https://api.ipify.org?format=json');
    let data = await res.json();
    return data.ip;
  } catch (e) { return '获取失败'; }
}

// 获取设备信息
function getDevice() {
  const ua = navigator.userAgent;
  if (ua.match(/iPhone|iPad/)) return '苹果设备 | ' + ua;
  if (ua.match(/Android/)) return '安卓设备 | ' + ua;
  if (ua.match(/Windows/)) return 'Windows电脑 | ' + ua;
  if (ua.match(/Mac/)) return 'Mac电脑 | ' + ua;
  return '未知设备 | ' + ua;
}

// 提交考试
async function submitExam() {
  const name = document.getElementById('name').value.trim();
  const phone = document.getElementById('phone').value.trim();

  if (!name) { alert('请填写姓名！'); return; }
  if (!phone) { alert('请填写手机号！'); return; }

  const ip = await getIp();
  const device = getDevice();
  const answers = document.querySelectorAll('.answer');
  let total = 0;
  let content = '📄 易慧家科技销售考试答卷\n';
  content += '=============================================\n';
  content += '👤 姓名：' + name + '\n';
  content += '📞 手机号：' + phone + '\n';
  content += '🌐 IP地址：' + ip + '\n';
  content += '📱 设备信息：' + device + '\n';
  content += '=============================================\n\n';

  const questions = document.querySelectorAll('.question p');
  for (let i = 0; i < questions.length; i++) {
    const qText = questions[i].innerText;
    const ansText = answers[i].value.trim() || '【未作答】';
    const score = answers[i].getAttribute('data-score') || 0;
    total += Number(score);
    content += qText + '\n▶ 答案：' + ansText + '\n\n';
  }

  content += '🎯 考试总分：' + total + ' 分（满分100）\n';
  const blob = new Blob([content], { type: 'text/plain' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = name + '_' + phone + '_易慧家销售考试答卷.txt';
  a.click();

  alert('提交成功！已自动下载答卷，请发送给管理员');
}
</script>
